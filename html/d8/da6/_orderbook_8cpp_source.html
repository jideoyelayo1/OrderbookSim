43Trades Orderbook::MatchOrders()
44{
45 Trades trades;
46 trades.reserve(_orders.size());
47
48 while (true)
49 {
50 if (_bids.empty() || _asks.empty())
51 break;
52
53 auto& [bidPrice, bids] = *_bids.begin();
54 auto& [askPrice, asks] = *_asks.begin();
55
56 if (bidPrice > askPrice) // for some reasoon this works: bidPrice > askPrice should be the other way around
57 break;
58
59 while (!bids.empty() && !asks.empty())
60 {
61 OrderPtr bid = bids.front();
62 OrderPtr ask = asks.front();
63
64 Quantity quantity = std::min(bid->getRemainingQty(), ask->getRemainingQty());
65
66 bid->Fill(quantity);
67 ask->Fill(quantity);
68
69 if (bid->isFilled())
70 {
71 bids.pop_front();
72 _orders.erase(bid->getOrderId());
73 }
74
75 if (ask->isFilled())
76 {
77 asks.pop_front();
78 _orders.erase(ask->getOrderId());
79 }
80
81 trades.push_back(Trade{
82 TradeInfo{ bid->getOrderId(), bid->getPrice(), quantity },
83 TradeInfo{ ask->getOrderId(), ask->getPrice(), quantity }
84 });
85
86 Price purchasedPrice = 0;
87
88 switch (_executionType) {
89 case ExecutionTypes::BuyersPrice:
90 purchasedPrice = std::max(ask->getPrice(), bid->getPrice());
91 break;
92 case ExecutionTypes::SellersPrice:
93 purchasedPrice = std::min(ask->getPrice(), bid->getPrice());
94 break;
95 case ExecutionTypes::MidPrice:
96 purchasedPrice = (ask->getPrice() + bid->getPrice()) / 2;
97 break;
98 default: break;
99 }
100
101 _orderDetailHistory.removeMatchedOrder(bid->getOrderId(), ask->getOrderId());
102
103 onOrderMatched(bid->getPrice(), quantity, bid->isFilled());
104 onOrderMatchedWithHistoryUpdate(ask->getPrice(), quantity, ask->isFilled(), purchasedPrice); // we are purchasing at the max price
105 }
106
107 if (bids.empty())
108 {
109 _bids.erase(bidPrice);
110 _data.erase(bidPrice);
111 }
112
113 if (asks.empty())
114 {
115 _asks.erase(askPrice);
116 _data.erase(askPrice);
117 }
118 }
119
120 if (!_bids.empty())
121 {
122 auto& [_, bids] = *_bids.begin();
123 auto& order = bids.front();
124 if (order->getOrderType() == OrderType::FillAndKill)
125 CancelOrder(order->getOrderId());
126 }
127
128 if (!_asks.empty())
129 {
130 auto& [_, asks] = *_asks.begin();
131 auto& order = asks.front();
132 if (order->getOrderType() == OrderType::FillAndKill)
133 CancelOrder(order->getOrderId());
134 }
135
136 return trades;
137}

189bool Orderbook::canFullyFill(Side side, Price price, Quantity quantity) const {
190 if (!canMatch(side, price)) return false;
191
192 std::optional<Price> threshold;
193
194 if (side == Side::Buy) {
195 const auto& [askPrice, _] = *_asks.begin();
196 threshold = askPrice;
197 }
198 else {
199 const auto& [bidPrice, _] = *_bids.begin();
200 threshold = bidPrice;
201 }
202 for (const auto& [levelPrice, levelData] : _data) {
203 if (threshold.has_value() &&
204 (side == Side::Buy && threshold.value() > levelPrice) ||
205 (side == Side::Sell && threshold.value() < levelPrice)
206 ) continue;
207 if ((side == Side::Buy && levelPrice > price) ||
208 (side == Side::Sell && levelPrice < price)) continue;
209 if (quantity <= levelData._quantity) return true;
210 quantity -= levelData._quantity;
211 }
212 return false;
213}

316void Orderbook::PruneGoodForDayOrders() {
317 using namespace std::chrono;
318 const auto end = hours(16);
319 while (true) {
320 const auto now = system_clock::now();
321 const auto now_c = system_clock::to_time_t(now);
322 std::tm now_parts;
323 localtime_s(&now_parts, &now_c);
324
325 if (now_parts.tm_hour >= end.count()) now_parts.tm_mday++;
326
327 now_parts.tm_hour = end.count();
328 now_parts.tm_min = 0;
329 now_parts.tm_sec = 0;
330
331 auto next = system_clock::from_time_t(mktime(&now_parts));
332 auto till = next - now + milliseconds(100);
333
334 { // we don't want this thread to alter the state of our data structures at the same time
335 std::unique_lock ordersLock{ _ordersMutex };
336 if (_shutdown.load(std::memory_order_acquire)
337 || _shutdownConditionVariable.wait_for(ordersLock, till) == std::cv_status::no_timeout)
338 return;
339 }
340
341 OrderIds orderIds;
342
343 {
344 std::scoped_lock ordersLock{ _ordersMutex };
345 for (const auto& [_, entry] : _orders) {
346 const auto& [order, _] = entry;
347 if (order->getOrderType() != OrderType::GoodForDay) continue;
348 orderIds.push_back(order->getOrderId());
349 }
350 }
351
352 CancelOrders(orderIds);
353 }
354}

412Trades Orderbook::addOrder(OrderPtr order) {
413 std::scoped_lock ordersLock{ _ordersMutex };
414
415 // Add to history
416 _orderDetailHistory.addOrderToHistory(order->getOrderType(), order->getOrderId(), order->getSide(), order->getPrice(), order->getQty());
417
418 if (_orders.contains(order->getOrderId())) return { };
419
420 if (order->getOrderType() == OrderType::Market)
421 {
422 if (order->getSide() == Side::Buy && !_asks.empty())
423 {
424 const auto& [worstAsk, _] = *_asks.rbegin();
425 order->toGoodTillCancel(worstAsk);
426 }
427 else if (order->getSide() == Side::Sell && !_bids.empty())
428 {
429 const auto& [worstBid, _] = *_bids.rbegin();
430 order->toGoodTillCancel(worstBid);
431 }
432 else
433 return { };
434 }
435
436 if (order->getOrderType() == OrderType::FillAndKill && !canMatch(order->getSide(), order->getPrice())) return { };
437 if (order->getOrderType() == OrderType::FillOrKill && !canFullyFill(order->getSide(), order->getPrice(), order->getInitialQty())) return { };
438
439 OrderPtrs::iterator iter;
440 if (order->getSide() == Side::Buy) {
441 auto& orders = _bids[order->getPrice()];
442 orders.push_back(order);
443 iter = std::prev(orders.end());
444 }
445 else {
446 auto& orders = _asks[order->getPrice()];
447 orders.push_back(order);
448 iter = std::prev(orders.end());
449 }
450 _orders.insert({ order->getOrderId(), OrderEntry{order,iter} });
451
452 onOrderAdded(order);
453
454 return MatchOrders();
455}

583void Orderbook::EndOfDay() {
584 std::unordered_map<OrderId, bool> goodToCancels;
585
586 // Handle live orders
587 for (const auto& m : _orderDetailHistory.getLiveOrders()) {
588 if (m.second.OrderType() != OrderType::GoodForDay) continue;
589 goodToCancels[m.first] = true;
590 _orderDetailHistory.deleteALiveOrder(m.first);
591 }
592
593 // Remove orders not in goodToCancels
594 for (auto it = _orders.begin(); it != _orders.end(); ) {
595 if (goodToCancels.find(it->first) != goodToCancels.end()) {
596 ++it;
597 }
598 else {
599 it = _orders.erase(it);
600 }
601 }
602
603 // Clean up bids
604 for (auto it = _bids.begin(); it != _bids.end(); ) {
605 auto& orderptrs = it->second;
606 for (auto orderIt = orderptrs.begin(); orderIt != orderptrs.end(); ) {
607 if (goodToCancels.find((*orderIt)->getOrderId()) != goodToCancels.end()) {
608 orderIt = orderptrs.erase(orderIt);
609 }
610 else {
611 ++orderIt;
612 }
613 }
614 if (orderptrs.empty()) {
615 it = _bids.erase(it);
616 }
617 else {
618 ++it;
619 }
620 }
621
622 // Clean up asks
623 for (auto it = _asks.begin(); it != _asks.end(); ) {
624 auto& orderptrs = it->second;
625 for (auto orderIt = orderptrs.begin(); orderIt != orderptrs.end(); ) {
626 if (goodToCancels.find((*orderIt)->getOrderId()) != goodToCancels.end()) {
627 orderIt = orderptrs.erase(orderIt);
628 }
629 else {
630 ++orderIt;
631 }
632 }
633 if (orderptrs.empty()) {
634 it = _asks.erase(it);
635 }
636 else {
637 ++it;
638 }
639 }
640}
