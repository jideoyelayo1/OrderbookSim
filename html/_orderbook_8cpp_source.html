15Trades Orderbook::MatchOrders()
16{

18 trades.reserve(_orders.size());

20 while (true)

22 if (_bids.empty() || _asks.empty())
23 break;
24
25 auto& [bidPrice, bids] = *_bids.begin();
26 auto& [askPrice, asks] = *_asks.begin();

29 break;

43 bids.pop_front();
44 _orders.erase(bid->getOrderId());
45 }

56 TradeInfo{ ask->getOrderId(), ask->getPrice(), quantity }

58
59 Price purchasedPrice = 0;
60
61 switch (_executionType) {
62 case ExecutionTypes::BuyersPrice:
63 purchasedPrice = std::max(ask->getPrice(), bid->getPrice());
64 break;
65 case ExecutionTypes::SellersPrice:
66 purchasedPrice = std::min(ask->getPrice(), bid->getPrice());
67 break;
68 case ExecutionTypes::MidPrice:
69 purchasedPrice = (ask->getPrice() + bid->getPrice()) / 2;
70 break;
71 default: break;
72 }
73
74 _orderDetailHistory.removeMatchedOrder(bid->getOrderId(), ask->getOrderId());
75
76 onOrderMatched(bid->getPrice(), quantity, bid->isFilled());
77 onOrderMatchedWithHistoryUpdate(ask->getPrice(), quantity, ask->isFilled(), purchasedPrice); // we are purchasing at the max price
78 }
79
80 if (bids.empty())
81 {
82 _bids.erase(bidPrice);
83 _data.erase(bidPrice);
84 }
85
86 if (asks.empty())
87 {
88 _asks.erase(askPrice);
89 _data.erase(askPrice);
90 }
91 }

93 if (!_bids.empty())

98 CancelOrder(order->getOrderId());

106 CancelOrder(order->getOrderId());
107 }
108

137bool Orderbook::canFullyFill(Side side, Price price, Quantity quantity) const { // if the order is fully filled we need to remove the order count from our bookkeeping DS, if not fully filled is does not need to be touched
138 if (!canMatch(side, price)) return false; // does this order cause a cross
139
140 std::optional<Price>threshold;
141
142 if (side == Side::Buy) { // if it can match and its a buy we know theres at least one sell
143 const auto& [askPrice, _] = *_asks.begin();
144 threshold = askPrice;
145
146 }
147 else {// if it can match and its a sell we know that bids exists
148 const auto& [bidPrice, _] = *_bids.begin();
149 threshold = bidPrice;
150 }
151 for (const auto& [levelPrice, levelData] : _data) { // we care about the quantity in the range of the best ask and the ask the person wants: filter for at or below that levelPrice
152 if (threshold.has_value() &&
153 (side == Side::Buy && threshold.value() > levelPrice) ||
154 (side == Side::Sell && threshold.value() < levelPrice)
155 ) continue;
156 if ((side == Side::Buy && levelPrice > price) ||
157 (side == Side::Sell && levelPrice < price)) continue;
158 //^ filtering in out everything we dont need first
159 if (quantity <= levelData._quantity) return true; // we're finding the acculate and seeing if we can find enough quantites
160 quantity -= levelData._quantity;
161 }
